        // ================================================================
        //  LABORATÓRIO (interativo)
        // ================================================================
        (function setupLab() {
            const aInput = document.getElementById('labA');
            const bInput = document.getElementById('labB');
            const opSelect = document.getElementById('labOp');
            const resultDiv = document.getElementById('labResult');

            function renderLab() {
                const a = parseFloat(aInput.value) || 0;
                const b = parseFloat(bInput.value) || 0;
                const op = opSelect.value;

                let steps = [];
                let final = '';

                const format = (x) => (Number.isInteger(x) ? x.toString() : x.toFixed(2));

                switch (op) {
                    case 'soma2': {
                        const a2 = a * a,
                            ab2 = 2 * a * b,
                            b2 = b * b;
                        steps.push(`(${format(a)} + ${format(b)})² = ${format(a)}² + 2·${format(a)}·${format(b)} + ${format(b)}²`);
                        steps.push(`= ${format(a2)} + ${format(ab2)} + ${format(b2)}`);
                        final = `= ${format(a2 + ab2 + b2)}`;
                        break;
                    }
                    case 'diff2': {
                        const a2 = a * a,
                            ab2 = 2 * a * b,
                            b2 = b * b;
                        steps.push(`(${format(a)} – ${format(b)})² = ${format(a)}² – 2·${format(a)}·${format(b)} + ${format(b)}²`);
                        steps.push(`= ${format(a2)} – ${format(ab2)} + ${format(b2)}`);
                        final = `= ${format(a2 - ab2 + b2)}`;
                        break;
                    }
                    case 'prod': {
                        const a2 = a * a,
                            b2 = b * b;
                        steps.push(`(${format(a)} + ${format(b)})·(${format(a)} – ${format(b)}) = ${format(a)}² – ${format(b)}²`);
                        steps.push(`= ${format(a2)} – ${format(b2)}`);
                        final = `= ${format(a2 - b2)}`;
                        break;
                    }
                    case 'soma3': {
                        const a3 = a * a * a,
                            a2b = 3 * a * a * b,
                            ab2 = 3 * a * b * b,
                            b3 = b * b * b;
                        steps.push(
                            `(${format(a)} + ${format(b)})³ = ${format(a)}³ + 3·${format(a)}²·${format(b)} + 3·${format(a)}·${format(b)}² + ${format(b)}³`
                            );
                        steps.push(`= ${format(a3)} + ${format(a2b)} + ${format(ab2)} + ${format(b3)}`);
                        final = `= ${format(a3 + a2b + ab2 + b3)}`;
                        break;
                    }
                    case 'diff3': {
                        const a3 = a * a * a,
                            a2b = 3 * a * a * b,
                            ab2 = 3 * a * b * b,
                            b3 = b * b * b;
                        steps.push(
                            `(${format(a)} – ${format(b)})³ = ${format(a)}³ – 3·${format(a)}²·${format(b)} + 3·${format(a)}·${format(b)}² – ${format(b)}³`
                            );
                        steps.push(`= ${format(a3)} – ${format(a2b)} + ${format(ab2)} – ${format(b3)}`);
                        final = `= ${format(a3 - a2b + ab2 - b3)}`;
                        break;
                    }
                }

                resultDiv.innerHTML = steps.map(s => `<div class="step">${s}</div>`).join('') +
                    `<div class="final">${final}</div>`;
            }

            aInput.addEventListener('input', renderLab);
            bInput.addEventListener('input', renderLab);
            opSelect.addEventListener('change', renderLab);
            renderLab();
        })();

            // ===== Classe Quiz =====
            class QuizApp {
                constructor(questions) {
                    this.rawQuestions = questions;
                    this.total = this.rawQuestions.length;
                    this.shuffledQuestions = [];
                    this.currentIndex = 0;
                    this.score = 0;
                    this.answered = false;
                    this.history = [];
                    this.startTime = Date.now();
                    this.timerInterval = null;
                    this.elapsedSeconds = 0;

                    this.container = document.getElementById('quizContainer');
                    this.renderInitial();
                }

                renderInitial() {
                    this.container.innerHTML = `
                        <div id="quizArea">
                            <div class="top-bar">
                                <div class="info">
                                    <span id="qCounter">1 / ${this.total}</span>
                                    <span id="scoreDisplay">✅ 0</span>
                                    <span id="categoryBadge" class="badge-category">Geral</span>
                                </div>
                                <div class="timer" id="timerDisplay">⏱️ 00:00</div>
                            </div>
                            <div class="progress-track"><div class="progress-fill" id="bar"></div></div>
                            <div id="question"></div>
                            <div id="options"></div>
                            <div id="feedback"></div>
                            <div class="actions">
                                <button class="btn" id="nextBtn" disabled>Próxima →</button>
                                <button class="btn btn-secondary" id="skipBtn">⏭ Pular</button>
                            </div>
                        </div>
                        <div id="resultArea" class="hidden">
                            <div class="text-center">
                                <span class="emoji-big" id="resultEmoji">🎉</span>
                                <h2 id="resultTitle">Treino concluído!</h2>
                                <p id="resultSub"></p>
                            </div>
                            <div class="stats-grid" id="statsGrid">
                                <div class="stat-item"><div class="stat-value" id="statAcertos">0</div><div class="stat-label">Acertos</div></div>
                                <div class="stat-item"><div class="stat-value" id="statErros">0</div><div class="stat-label">Erros</div></div>
                                <div class="stat-item"><div class="stat-value" id="statPct">0%</div><div class="stat-label">Aproveitamento</div></div>
                                <div class="stat-item"><div class="stat-value" id="statTempo">00:00</div><div class="stat-label">Tempo total</div></div>
                            </div>
                            <button class="btn btn-success" id="showReviewBtn">📖 Ver revisão completa</button>
                            <button class="btn" id="restartQuizBtn" style="margin-top:12px;">🔄 Recomeçar</button>
                            <div id="reviewArea" class="hidden"></div>
                        </div>
                    `;

                    // Referências
                    this.quizArea = document.getElementById('quizArea');
                    this.resultArea = document.getElementById('resultArea');
                    this.questionEl = document.getElementById('question');
                    this.optionsEl = document.getElementById('options');
                    this.feedbackEl = document.getElementById('feedback');
                    this.nextBtn = document.getElementById('nextBtn');
                    this.skipBtn = document.getElementById('skipBtn');
                    this.qCounter = document.getElementById('qCounter');
                    this.scoreDisplay = document.getElementById('scoreDisplay');
                    this.categoryBadge = document.getElementById('categoryBadge');
                    this.bar = document.getElementById('bar');
                    this.timerDisplay = document.getElementById('timerDisplay');

                    // Resultado
                    this.resultEmoji = document.getElementById('resultEmoji');
                    this.resultSub = document.getElementById('resultSub');
                    this.statAcertos = document.getElementById('statAcertos');
                    this.statErros = document.getElementById('statErros');
                    this.statPct = document.getElementById('statPct');
                    this.statTempo = document.getElementById('statTempo');
                    this.showReviewBtn = document.getElementById('showReviewBtn');
                    this.reviewArea = document.getElementById('reviewArea');
                    this.restartBtn = document.getElementById('restartQuizBtn');

                    // Eventos
                    this.nextBtn.addEventListener('click', () => this.nextQuestion());
                    this.skipBtn.addEventListener('click', () => this.skipQuestion());
                    this.showReviewBtn.addEventListener('click', () => this.toggleReview());
                    this.restartBtn.addEventListener('click', () => this.restart());

                    // Inicializa
                    this.shuffleAll();
                    this.startTimer();
                    this.renderQuestion();
                }

                shuffleArray(arr) {
                    for (let i = arr.length - 1; i > 0; i--) {
                        const j = Math.floor(Math.random() * (i + 1));
                        [arr[i], arr[j]] = [arr[j], arr[i]];
                    }
                    return arr;
                }

                shuffleAll() {
                    this.shuffledQuestions = this.shuffleArray([...this.rawQuestions]);
                    this.shuffledQuestions.forEach(q => {
                        const opts = q.options.map((text, idx) => ({ text, idx }));
                        const shuffledOpts = this.shuffleArray(opts);
                        q.options = shuffledOpts.map(o => o.text);
                        q.correct = shuffledOpts.findIndex(o => o.idx === q.correct);
                    });
                }

                startTimer() {
                    if (this.timerInterval) clearInterval(this.timerInterval);
                    this.startTime = Date.now();
                    this.timerInterval = setInterval(() => {
                        this.elapsedSeconds = Math.floor((Date.now() - this.startTime) / 1000);
                        this.timerDisplay.textContent = `⏱️ ${this.formatTime(this.elapsedSeconds)}`;
                    }, 500);
                }

                formatTime(sec) {
                    const m = String(Math.floor(sec / 60)).padStart(2, '0');
                    const s = String(sec % 60).padStart(2, '0');
                    return `${m}:${s}`;
                }

                renderQuestion() {
                    const q = this.shuffledQuestions[this.currentIndex];
                    this.questionEl.textContent = `${this.currentIndex+1}. ${q.q}`;
                    this.qCounter.textContent = `${this.currentIndex+1} / ${this.total}`;
                    this.scoreDisplay.textContent = `✅ ${this.score}`;
                    this.categoryBadge.textContent = this.getCategoryLabel(q.category);
                    this.bar.style.width = `${(this.currentIndex / this.total) * 100}%`;

                    this.optionsEl.innerHTML = '';
                    this.feedbackEl.textContent = '';
                    this.feedbackEl.className = '';
                    this.nextBtn.disabled = true;
                    this.answered = false;

                    q.options.forEach((text, idx) => {
                        const btn = document.createElement('button');
                        btn.className = 'option';
                        btn.textContent = text;
                        btn.dataset.idx = idx;
                        btn.addEventListener('click', () => this.selectOption(idx));
                        this.optionsEl.appendChild(btn);
                    });
                }

                getCategoryLabel(cat) {
                    const map = { 'S': 'Soma²', 'D': 'Diferença²', 'P': 'Soma·Dif', 'C+': 'Cubo+', 'C-': 'Cubo-',
                        'M': 'Misto' };
                    return map[cat] || 'Geral';
                }

                selectOption(selectedIdx) {
                    if (this.answered) return;
                    this.answered = true;

                    const q = this.shuffledQuestions[this.currentIndex];
                    const correctIdx = q.correct;
                    const isCorrect = (selectedIdx === correctIdx);
                    const buttons = this.optionsEl.querySelectorAll('.option');

                    buttons.forEach(btn => btn.disabled = true);
                    buttons.forEach((btn, idx) => {
                        if (idx === correctIdx) btn.classList.add('correct');
                        if (idx === selectedIdx && !isCorrect) btn.classList.add('wrong');
                    });

                    if (isCorrect) {
                        this.score++;
                        this.scoreDisplay.textContent = `✅ ${this.score}`;
                        this.feedbackEl.textContent = '✅ Correta! ';
                        this.feedbackEl.className = 'correct-feedback';
                    } else {
                        this.feedbackEl.textContent = '❌ Incorreta. ';
                        this.feedbackEl.className = 'wrong-feedback';
                    }
                    const expl = document.createElement('span');
                    expl.className = 'explanation';
                    expl.textContent = q.explanation;
                    this.feedbackEl.appendChild(expl);

                    this.history.push({
                        userAnswer: selectedIdx,
                        correct: correctIdx,
                        isCorrect,
                        timeSpent: this.elapsedSeconds
                    });

                    this.nextBtn.disabled = false;
                }

                nextQuestion() {
                    this.currentIndex++;
                    if (this.currentIndex < this.total) {
                        this.renderQuestion();
                    } else {
                        this.finishQuiz();
                    }
                }

                skipQuestion() {
                    if (this.answered) return;
                    const q = this.shuffledQuestions[this.currentIndex];
                    const buttons = this.optionsEl.querySelectorAll('.option');
                    buttons.forEach((btn, idx) => {
                        btn.disabled = true;
                        if (idx === q.correct) btn.classList.add('correct');
                    });
                    this.feedbackEl.textContent = '⏭ Pular: a resposta correta está marcada.';
                    this.feedbackEl.className = 'wrong-feedback';
                    const expl = document.createElement('span');
                    expl.className = 'explanation';
                    expl.textContent = q.explanation;
                    this.feedbackEl.appendChild(expl);
                    this.answered = true;
                    this.history.push({
                        userAnswer: -1,
                        correct: q.correct,
                        isCorrect: false,
                        timeSpent: this.elapsedSeconds
                    });
                    this.nextBtn.disabled = false;
                }

                finishQuiz() {
                    clearInterval(this.timerInterval);
                    this.quizArea.classList.add('hidden');
                    this.resultArea.classList.remove('hidden');

                    const total = this.total;
                    const acertos = this.score;
                    const erros = total - acertos;
                    const pct = Math.round((acertos / total) * 100);

                    let emoji, msg;
                    if (pct === 100) { emoji = '🌟';
                        msg = 'Perfeito! Você é um(a) mestre!'; } else if (pct >= 80) { emoji = '👏';
                        msg = 'Excelente! Continue assim.'; } else if (pct >= 60) { emoji = '📚';
                        msg = 'Bom! Reveja os erros.'; } else if (pct >= 40) { emoji = '💪';
                        msg = 'Pode melhorar! Estude mais.'; } else { emoji = '🔁';
                        msg = 'Não desanime! Refaça o treino.'; }

                    this.resultEmoji.textContent = emoji;
                    this.resultSub.textContent = msg;
                    this.statAcertos.textContent = acertos;
                    this.statErros.textContent = erros;
                    this.statPct.textContent = `${pct}%`;
                    this.statTempo.textContent = this.formatTime(this.elapsedSeconds);

                    this.reviewArea.classList.add('hidden');
                    this.showReviewBtn.textContent = '📖 Ver revisão completa';
                    this.reviewData = this.shuffledQuestions.map((q, idx) => ({
                        ...q,
                        userAnswer: this.history[idx] ? this.history[idx].userAnswer : -1,
                        isCorrect: this.history[idx] ? this.history[idx].isCorrect : false,
                    }));
                }

                toggleReview() {
                    if (this.reviewArea.classList.contains('hidden')) {
                        this.buildReview();
                        this.reviewArea.classList.remove('hidden');
                        this.showReviewBtn.textContent = '🙈 Ocultar revisão';
                    } else {
                        this.reviewArea.classList.add('hidden');
                        this.showReviewBtn.textContent = '📖 Ver revisão completa';
                    }
                }

                buildReview() {
                    this.reviewArea.innerHTML = '';
                    this.reviewData.forEach((q, idx) => {
                        const div = document.createElement('div');
                        div.className = 'review-item';
                        const userAns = q.userAnswer;
                        const isCorrect = q.isCorrect;
                        const correctText = q.options[q.correct];

                        let answerHtml = `<div class="r-answer">`;
                        if (userAns === -1) {
                            answerHtml +=
                                `<span class="r-wrong">Pulada</span> → Correta: <span class="r-correct">${correctText}</span>`;
                        } else if (isCorrect) {
                            answerHtml += `<span class="r-correct">✓ ${q.options[userAns]}</span> (correta)`;
                        } else {
                            answerHtml +=
                                `<span class="r-wrong">✗ ${q.options[userAns]}</span> → Correta: <span class="r-correct">${correctText}</span>`;
                        }
                        answerHtml += `</div>`;

                        div.innerHTML = `
                            <div class="r-question">${idx+1}. ${q.q}</div>
                            ${answerHtml}
                            <div class="r-explanation">📘 ${q.explanation}</div>
                        `;
                        this.reviewArea.appendChild(div);
                    });
                }

                restart() {
                    clearInterval(this.timerInterval);
                    this.currentIndex = 0;
                    this.score = 0;
                    this.history = [];
                    this.elapsedSeconds = 0;
                    this.answered = false;

                    this.shuffleAll();
                    this.quizArea.classList.remove('hidden');
                    this.resultArea.classList.add('hidden');
                    this.reviewArea.classList.add('hidden');
                    this.showReviewBtn.textContent = '📖 Ver revisão completa';

                    this.startTimer();
                    this.renderQuestion();
                }
            }

            // Instancia o Quiz quando a aba for ativada pela primeira vez
            let quizInstance = null;

            function initQuiz() {
                if (!quizInstance) {
                    quizInstance = new QuizApp(QUESTION_BANK);
                }
            }

            // Observa a aba de quiz para iniciar quando for clicada
            const quizTab = document.querySelector('[data-tab="quiz"]');
            const observer = new MutationObserver(() => {
                if (quizTab.classList.contains('active')) {
                    initQuiz();
                }
            });
            observer.observe(quizTab, { attributes: true, attributeFilter: ['class'] });

            // Se a aba já estiver ativa ao carregar (não é o caso), inicia
            // mas como a padrão é 'teoria', não precisa.
            // Para segurança, se o usuário clicar em quiz, o observer chama.
        })();


        // ================================================================
        //  NAVEGAÇÃO POR TABS
        // ================================================================
        document.querySelectorAll('.tab-btn').forEach(btn => {
            btn.addEventListener('click', function() {
                // Remove active de todos
                document.querySelectorAll('.tab-btn').forEach(b => b.classList.remove('active'));
                document.querySelectorAll('.tab-content').forEach(c => c.classList.remove('active'));

                // Ativa este
                this.classList.add('active');
                const tabId = this.dataset.tab;
                document.getElementById('tab-' + tabId).classList.add('active');

                // Se for a aba quiz, garante que o quiz seja iniciado (já tratado pelo observer)
                if (tabId === 'quiz') {
                    // O observer já fará, mas forçamos caso não tenha sido capturado
                    if (typeof initQuiz !== 'undefined') {
                        try { initQuiz(); } catch (e) {}
                    }
                }
            });
        });
    </script>
